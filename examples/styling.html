Styling

It's possible to style the magic ring. 

    // Set Default Styles
    var magic = new MagicCircle("#repeating", 2);
        magic.styles.colors.ring = "#973b67";
        magic.styles.colors.text = "#69a33c";
        magic.styles.colors.smallRing = "#3c71af";

    // Dynamic Recoloring
        magic.cast()
          .ring()
             .color("#F00")
          .text(16,"BAR")
             .color("#BA5")
