Interactivity

You can listen for any interactivity event on a ring element.

  // draw text
  var text = interact.cast()
      .text(25, "Hover over me!")
      .getLast();
  // now backspace over it and draw a sensor ring
    interact.cast()
      .backspace(25)
      .ring(25)
      .color("useNone")
      .on("mouseenter", function rotateText() {
        text.rotation(-35);
    })
}
